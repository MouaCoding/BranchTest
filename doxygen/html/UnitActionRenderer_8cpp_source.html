97  bool IsFirst = true;
98  bool Moveable = true;
99  bool HasCargo = false;
100  EAssetType UnitType;
101 
102  for(auto &Command : DDisplayedCommands){
103  Command = actNone;
104  }
105  if(!selectionlist.size()){
106  return;
107  }
108  for(auto &Iterator : selectionlist){
109  if(auto Asset = Iterator.lock()){
110  if(DPlayerColor != Asset->Color()){
111  return;
112  }
113  if(IsFirst){
114  UnitType = Asset->Type();
115  IsFirst = false;
116  Moveable = 0 < Asset->Speed();
117  }
118  else if(UnitType != Asset->Type()){
119  AllSame = false;
120  }
121  if(Asset->Lumber() || Asset->Gold()){
122  HasCargo = true;
123  }
124  }
125  }
126  if(actNone == currentaction){
127  if(Moveable){
128 
129  DDisplayedCommands[0] = HasCargo ? actConvey : actMove;
130  DDisplayedCommands[1] = actStandGround;
131  DDisplayedCommands[2] = actAttack;
132  if(auto Asset = selectionlist.front().lock()){
133  if(Asset->HasCapability(actRepair)){
134  DDisplayedCommands[3] = actRepair;
135  }
136  if(Asset->HasCapability(actPatrol)){
137  DDisplayedCommands[3] = actPatrol;
138  }
139  if(Asset->HasCapability(actMine)){
140  DDisplayedCommands[4] = actMine;
141  }
142  if(Asset->HasCapability(actBuildSimple) && (1 == selectionlist.size())){
143  DDisplayedCommands[6] = actBuildSimple;
144  }
145  }
146  }
147  else{
148  if(auto Asset = selectionlist.front().lock()){
149  if((aaConstruct == Asset->Action())||(aaCapability == Asset->Action())){
150  DDisplayedCommands.back() = actCancel;
151  }
152  else{
153  int Index = 0;
154  for(auto &Capability : Asset->Capabilities()){
155  DDisplayedCommands[Index] = Capability;

157  if(DDisplayedCommands.size() <= Index){

162  }
163  }
164  }
165  else if(actBuildSimple == currentaction){
