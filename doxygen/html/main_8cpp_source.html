532  RenderButtonMenuMode();
533  break;
534  case gmSoundOptions:
535  case gmNetworkOptions: ProcessInputEditOptionsMode();
536  CalculateEditOptionsMode();
537  RenderEditOptionsMode();
538  break;
539  case gmMapSelect: ProcessInputMapSelectMode();
540  CalculateMapSelectMode();

542  break;
543  case gmPlayerAISelect: ProcessInputPlayerAISelectMode();
544  CalculatePlayerAISelectMode();
545  RenderPlayerAISelectMode();
546  break;
547  case gmBattle:
548  default: DGameModel->ClearGameEvents();
549  ProcessInputGameMode();
550  CalculateGameMode();
551  RenderGameMode();
552  break;
553  }
554 
555  for(int Index = 0; Index < pcMax; Index++){
556  if(!DLeftDown[Index] || (2 != DLeftClick[Index])){
557  DLeftClick[Index] = 0;
558  }
559  DRightClick[Index] = 0;
560  }
561 
562  DCursorSet->DrawCursor(DWorkingBufferPixmap, DDrawingContext, DCurrentX[DPlayerColor], DCurrentY[DPlayerColor], DCursorIndices[DCursorType]);

567  if(DNextGameMode != DGameMode){
568  SwitchButtonMenuData();
569  }
570  DGameMode = DNextGameMode;
571 
572  return TRUE;
573 }
