107 bool CPath::DecomposePath(std::vector< std::string > &decomp, const std::string &path){
108  size_t Anchor, DelimiterLocation;
109  bool PathIsRelative = true;
110  size_t Index = 0;
111 
112  if(0 == path.length()){
113  return false;
114  }
115  if(path[0] == DDelimiter){
116  PathIsRelative = false;
117  }
118  decomp.clear();
119  Anchor = 0;
120  while((Anchor + 1 < path.length()) && (-1 != (DelimiterLocation = path.find(DDelimiterString, Anchor)))){
121  decomp.push_back(path.substr(Anchor, DelimiterLocation - Anchor));
122  Anchor = DelimiterLocation + 1;
123  }
124  if(Anchor < path.length()){
125  std::string TempString = path.substr(Anchor, path.length() - Anchor);
126 
127  if(TempString != DDelimiterString){
128  decomp.push_back(TempString);
129  }
130  }
131  if(decomp.size()){
132  if(!decomp[0].length()){
133  decomp.erase(decomp.begin());
134  }
135  }
136  while(Index < decomp.size()){
137  if(decomp[Index] == "."){
138  decomp.erase(decomp.begin());
139  }
140  else if(decomp[Index] == ".."){
141  if(Index){
142  Index--;
143  decomp.erase(decomp.begin() + Index);
144  decomp.erase(decomp.begin() + Index);
145  }
146  else if(!PathIsRelative){
147  return false;
148  }
149  else{
150  Index++;
151  }
152  }
153  else{
154  Index++;
155  }
156  }
157 
158  return true;
159 }
