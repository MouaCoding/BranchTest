55 std::string CButtonRenderer::Text(const std::string &text, bool minimize){
56  gint TotalWidth, TotalHeight, Top, Bottom;
57  DText = text;
58  DFont->MeasureTextDetailed(DText, TotalWidth, TotalHeight, Top, Bottom);
59 
60  TotalHeight = Bottom - Top + 1;
61  if(TotalHeight + DOuterBevel->Width() * 2 > DHeight){
62  DHeight = TotalHeight + DOuterBevel->Width() * 2;
63  }
64  else if(minimize){
65  DHeight = TotalHeight + DOuterBevel->Width() * 2;
66  }
67  if(TotalWidth + DOuterBevel->Width() * 2 > DWidth){
68  DWidth = TotalWidth + DOuterBevel->Width() * 2;
69  }
70  else if(minimize){
71  DWidth = TotalWidth + DOuterBevel->Width() * 2;
72  }
73  DTextOffsetX = DWidth/2 - TotalWidth/2;
74  DTextOffsetY = DHeight/2 - TotalHeight/2 - Top;
75  return DText;

102 void CButtonRenderer::DrawButton(GdkDrawable *drawable, int x, int y, EButtonState state){
103  GdkGC *TempGC = gdk_gc_new(drawable);
104 
105  if(bsPressed == state){
106  int BevelWidth = DInnerBevel->Width();
107  DColorTileset->DrawTileRectangle(drawable, x, y, DWidth, DHeight, DDarkIndices[DButtonColor]);
108  DFont->DrawTextWithShadow(drawable, TempGC, x + DTextOffsetX, y + DTextOffsetY, DWhiteIndex, DBlackIndex, 1, DText);
109  DInnerBevel->DrawBevel(drawable, x + BevelWidth, y + BevelWidth, DWidth - BevelWidth * 2, DHeight - BevelWidth * 2);
110  }
111  else if(bsInactive == state){
112  int BevelWidth = DOuterBevel->Width();
113  DColorTileset->DrawTileRectangle(drawable, x, y, DWidth, DHeight, DDarkIndices[pcBlack]);
114  DFont->DrawTextWithShadow(drawable, TempGC, x + DTextOffsetX, y + DTextOffsetY, DBlackIndex, DWhiteIndex, 1, DText);
115  DOuterBevel->DrawBevel(drawable, x + BevelWidth, y + BevelWidth, DWidth - BevelWidth * 2, DHeight - BevelWidth * 2);
116  }
117  else{
118  int BevelWidth = DOuterBevel->Width();
119  DColorTileset->DrawTileRectangle(drawable, x, y, DWidth, DHeight, DLightIndices[DButtonColor]);
120  DFont->DrawTextWithShadow(drawable, TempGC, x + DTextOffsetX, y + DTextOffsetY, bsHover == state ? DWhiteIndex : DGoldIndex, DBlackIndex, 1, DText);
121  DOuterBevel->DrawBevel(drawable, x + BevelWidth, y + BevelWidth, DWidth - BevelWidth * 2, DHeight - BevelWidth * 2);
122  }
123 
124  g_object_unref(TempGC);
125 }
