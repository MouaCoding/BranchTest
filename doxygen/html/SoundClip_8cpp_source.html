141 bool CSoundClip::Load(std::shared_ptr< CDataSource > source){
142  CSFVirtualIODataSource VIODataSource(source);
143  SF_VIRTUAL_IO SFVirtualIO = VIODataSource.SFVirtualIO();
144  SNDFILE *SoundFilePtr;
145  SF_INFO SoundFileInfo;
146 
147  //SoundFilePtr = sf_open(filename.c_str(), SFM_READ, &SoundFileInfo);
148  SoundFilePtr = sf_open_virtual(&SFVirtualIO, SFM_READ, &SoundFileInfo, (void *)&VIODataSource);
149  if(NULL == SoundFilePtr){
150  return false;
151  }
152  if(1 == SoundFileInfo.channels){
153  DChannels = 2;
154  DTotalFrames = SoundFileInfo.frames;
155  DSampleRate = SoundFileInfo.samplerate;
156  DData = new float[SoundFileInfo.frames * 2];
157  for(int Frame = 0; Frame < DTotalFrames; Frame++){
158  sf_readf_float(SoundFilePtr, DData + Frame * 2, 1);
159  DData[Frame * 2 + 1] = DData[Frame * 2];
160  }
161  }
162  else if(2 == SoundFileInfo.channels){
163  DChannels = 2;
164  DTotalFrames = SoundFileInfo.frames;
165  DSampleRate = SoundFileInfo.samplerate;
166  DData = new float[SoundFileInfo.frames * SoundFileInfo.channels];
167  sf_readf_float(SoundFilePtr, DData, DTotalFrames);
168  }
169  else{
170  sf_close(SoundFilePtr);
171  return false;
172  }
173 
174  sf_close(SoundFilePtr);
175 
176  return true;
