308 void CMapRenderer::MakeHammingSet(int value, std::vector< int > &hammingset){
309  int BitCount, Anchor, LastEnd;
310  hammingset.clear();
311  for(int Index = 0; Index < 8; Index++){
312  int Value = 1<<Index;
313  if(value & Value){
314  hammingset.push_back(Value);
315  }
316  }
317  LastEnd = BitCount = hammingset.size();
318  Anchor = 0;
319  for(int TotalBits = 1; TotalBits < BitCount; TotalBits++){
320  for(int LastIndex = Anchor; LastIndex < LastEnd; LastIndex++){
321  for(int BitIndex = 0; BitIndex < BitCount; BitIndex++){
322  int NewValue = hammingset[LastIndex] | hammingset[BitIndex];
323  if(NewValue != hammingset[LastIndex]){
324  bool Found = false;
325  for(int Index = LastEnd; Index < hammingset.size(); Index++){
326  if(NewValue == hammingset[Index]){
327  Found = true;
328  break;
329  }
330  }
331  if(!Found){
332  hammingset.push_back(NewValue);
333  }
334  }
335  }
336  }
337  Anchor = LastEnd + 1;
338  LastEnd = hammingset.size();
339  }
340 }

349  std::vector< int > HammingSet;
350  MakeHammingSet(unknown, HammingSet);
