99 void CEditRenderer::DrawEdit(GdkDrawable *drawable, int x, int y, int cursorpos){
100  GdkGC *TempGC = gdk_gc_new(drawable);
101  int BevelWidth = DInnerBevel->Width();
102  int TextColorIndex;
103  gint TextWidth, TextHeight;
104  std::string RenderText;
105 
106  DColorTileset->DrawTileRectangle(drawable, x, y, DWidth, DHeight, 0 <= cursorpos ? DLightIndices[DBackgroundColor] : DDarkIndices[DBackgroundColor]);
107  if(0 <= cursorpos){
108  if(DText.length()){
109  bool RemoveCharacters = false;
110  RenderText = DText.substr(0,cursorpos);
111  RenderText += "|";
112  do{
113  DFont->MeasureText(RenderText, TextWidth, TextHeight);
114  TextWidth += BevelWidth * 2;
115  if(TextWidth > DWidth){
116  RenderText = RenderText.substr(1, RenderText.length()-1);
117  RemoveCharacters = true;
118  }
119  }while(DWidth < TextWidth);
120  if(!RemoveCharacters){
121  if(cursorpos < DText.length()){
122  RenderText += DText.substr(cursorpos);
123  }
124  }
125  }
126  else{
127  RenderText = "|";
128  }
129  }
130  else{
131  RenderText = DText;
132  }
133  do{
134  DFont->MeasureText(RenderText, TextWidth, TextHeight);
135  TextWidth += BevelWidth * 2;
136  if(TextWidth > DWidth){
137  RenderText = RenderText.substr(0, RenderText.length()-1);
138  }
139  }while(DWidth < TextWidth);
140  if(DTextValid){
141  TextColorIndex = 0 <= cursorpos ? DWhiteIndex : DGoldIndex;
142  }
143  else{
144  TextColorIndex = DRedIndex;
145  }
146  DFont->DrawTextWithShadow(drawable, TempGC, x + BevelWidth, y + BevelWidth, TextColorIndex, DBlackIndex, 1, RenderText);
147  DInnerBevel->DrawBevel(drawable, x + BevelWidth, y + BevelWidth, DWidth - BevelWidth * 2, DHeight - BevelWidth * 2);
148  g_object_unref(TempGC);
149 }
