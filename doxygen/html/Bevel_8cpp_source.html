57 void CBevel::DrawBevel(GdkDrawable *drawable, gint xpos, gint ypos, gint width, gint height){
58  GdkGC *TempGC = gdk_gc_new(drawable);
59  int TopY = ypos - DWidth;
60  int BottomY = ypos + height;
61  int LeftX = xpos - DWidth;
62  int RightX = xpos + width;
63 
64  DTileset->DrawTile(drawable, TempGC, LeftX, TopY, DCornerIndices[0]);
65  DTileset->DrawTile(drawable, TempGC, RightX, TopY, DCornerIndices[1]);
66  DTileset->DrawTile(drawable, TempGC, LeftX, BottomY, DCornerIndices[2]);
67  DTileset->DrawTile(drawable, TempGC, RightX, BottomY, DCornerIndices[3]);
68 
69  for(int XOff = 0; XOff < width; XOff += DWidth){
70  int Index = 0;
71  if(XOff + DWidth > width){
72  Index = width - XOff;
73  }
74  DTileset->DrawTile(drawable, TempGC, xpos + XOff, TopY, DTopIndices[Index]);
75  DTileset->DrawTile(drawable, TempGC, xpos + XOff, BottomY, DBottomIndices[Index]);
76  }
77  for(int YOff = 0; YOff < height; YOff += DWidth){
78  int Index = 0;
79  if(YOff + DWidth > height){
80  Index = height - YOff;
81  }
82  DTileset->DrawTile(drawable, TempGC, LeftX, ypos + YOff, DLeftIndices[Index]);
83  DTileset->DrawTile(drawable, TempGC, RightX, ypos + YOff, DRightIndices[Index]);
84  }
85  g_object_unref(TempGC);
86 }
