32 bool CFontTileset::LoadFont(std::shared_ptr< CDataSource > source){
33  CLineDataSource LineSource(source);
34  std::string TempString;
35  bool ReturnStatus = false;
36 
37  if(!LoadTileset(source)){
38  return false;
39  }
40 
41  DCharacterWidths.resize(DTileCount);
42  DDeltaWidths.resize(DTileCount);
43  DCharacterTops.resize(DTileCount);
44  DCharacterBottoms.resize(DTileCount);
45  DCharacterBaseline = DTileHeight;
46  try{
47  for(int Index = 0; Index < DTileCount; Index++){
48  if(!LineSource.Read(TempString)){
49  goto LoadFontExit;
50  }
51  DCharacterWidths[Index] = std::stoi(TempString);
52  }
53  for(int FromIndex = 0; FromIndex < DTileCount; FromIndex++){
54  std::vector< std::string > Values;
55  DDeltaWidths[FromIndex].resize(DTileCount);
56  if(!LineSource.Read(TempString)){
57  goto LoadFontExit;
58  }
59  CTokenizer::Tokenize(Values, TempString);
60  if(Values.size() != DTileCount){
61  goto LoadFontExit;
62  }
63  for(int ToIndex = 0; ToIndex < DTileCount; ToIndex++){
64  DDeltaWidths[FromIndex][ToIndex] = std::stoi(Values[ToIndex]);
65  }
66  }
67  ReturnStatus = true;
68  }
69  catch(std::exception &E){
70  printf("%s\n",E.what());
71  }
72  DPixbufTilesets.clear();
73  DPixbufTilesets.push_back(DPixbufTileset);
74  if(DPixelColors.size()){
75  int Width, Height, NumberChannels, RowStride;
76  guchar SrcRed, SrcGreen, SrcBlue;
77  std::vector< gint > BottomOccurence;
78  int BestLine = 0;
79 
80  Height = gdk_pixbuf_get_height(DPixbufTileset);
81  Width = gdk_pixbuf_get_width(DPixbufTileset);
82  NumberChannels = gdk_pixbuf_get_n_channels(DPixbufTileset);
83  RowStride = gdk_pixbuf_get_rowstride(DPixbufTileset);
84 
85  SrcRed = (DPixelColors[0].pixel>>16) & 0xFF;
86  SrcGreen = (DPixelColors[0].pixel>>8) & 0xFF;
87  SrcBlue = DPixelColors[0].pixel & 0xFF;
88  BottomOccurence.resize(DTileHeight + 1);
89  for(int Index = 0; Index < BottomOccurence.size(); Index++){
90  BottomOccurence[Index] = 0;
91  }
92  for(int Index = 0; Index < DTileCount; Index++){
93  int TopOpaque = DTileHeight, BottomOpaque = 0;
94  for(int Row = 0; Row < DTileHeight; Row++){
95  bool RowClear = true;
96  guchar *Pixel = gdk_pixbuf_get_pixels(DPixbufTileset) + (DTileHeight * Index + Row) * RowStride;
97  for(int Column = 0; Column < DTileWidth; Column++){
98  if(Pixel[3]){
99  RowClear = false;
100  break;
101  }
102  Pixel += NumberChannels;
103  }
104  if(!RowClear){
105  if(Row < TopOpaque){
106  TopOpaque = Row;
107  }
108  BottomOpaque = Row;
109  }
110  }
111  DCharacterTops[Index] = TopOpaque;
112  DCharacterBottoms[Index] = BottomOpaque;
113  BottomOccurence[BottomOpaque]++;
114  }
115  for(int Index = 1; Index < BottomOccurence.size(); Index++){
116  if(BottomOccurence[BestLine] < BottomOccurence[Index]){
117  BestLine = Index;
118  }
119  }
120  DCharacterBaseline = BestLine;
121  for(int Index = 1; Index < DPixelColors.size(); Index++){
122  guchar DestRed, DestGreen, DestBlue;
123 
124  DestRed = (DPixelColors[Index].pixel>>16) & 0xFF;
125  DestGreen = (DPixelColors[Index].pixel>>8) & 0xFF;
126  DestBlue = DPixelColors[Index].pixel & 0xFF;
127  DPixbufTilesets.push_back(gdk_pixbuf_copy(DPixbufTileset));
128 
129  for(int YPos = 0; YPos < Height; YPos++){
130  for(int XPos = 0; XPos < Width; XPos++){
131  guchar *Pixel = gdk_pixbuf_get_pixels(DPixbufTilesets.back()) + YPos * RowStride + XPos * NumberChannels;
132  if((SrcRed == Pixel[0])&&(SrcGreen == Pixel[1])&&(SrcBlue == Pixel[2])){
133  Pixel[0] = DestRed;
134  Pixel[1] = DestGreen;
135  Pixel[2] = DestBlue;
136  }
137  }
138  }
139  }
140  }
141 
142 LoadFontExit:
143  return ReturnStatus;
144 }

192 void CFontTileset::MeasureTextDetailed(const std::string &str, gint &width, gint &height, gint &top, gint &bottom){
193  int LastChar, NextChar;
194  width = 0;
195  top = DTileHeight;
196  bottom = 0;
197  for(int Index = 0; Index < str.length(); Index++){
198  NextChar = str[Index] - ' ';
199 
200  if(Index){
201  width += DDeltaWidths[LastChar][NextChar];
202  }
203  width += DCharacterWidths[NextChar];
204  if(DCharacterTops[NextChar] < top){
205  top = DCharacterTops[NextChar];
206  }
207  if(DCharacterBottoms[NextChar] > bottom){
208  bottom = DCharacterBottoms[NextChar];
209  }
210  LastChar = NextChar;
211  }
212  height = DTileHeight;
213 }
