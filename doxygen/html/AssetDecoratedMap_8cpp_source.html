181  int BestDistance = -1, CurDistance;
182  CPosition BestPosition(-1, -1);
183  TopY = fromasset->TilePositionY() - placeasset->Size();
184  BottomY = fromasset->TilePositionY() + fromasset->Size();
185  LeftX = fromasset->TilePositionX() - placeasset->Size();
186  RightX = fromasset->TilePositionX() + fromasset->Size();
187 
188  while(true){
189  int Skipped = 0;
190  if(0 <= TopY){
191  int ToX = MIN(RightX, Width() - 1);
192  for(int CurX = MAX(LeftX, 0); CurX <= ToX; CurX++){
193  if(CanPlaceAsset(CPosition(CurX, TopY), placeasset->Size(), placeasset)){
194  CPosition TempPosition(CurX, TopY);
195  CurDistance = TempPosition.DistanceSquared(nexttiletarget);
196  if((-1 == BestDistance)||(CurDistance < BestDistance)){
197  BestDistance = CurDistance;
198  BestPosition = TempPosition;
199  }
200  }
201  }
202  }
203  else{
204  Skipped++;
205  }
206  if(Width() > RightX){
207  int ToY = MIN(BottomY, Height() - 1);
208  for(int CurY = MAX(TopY, 0); CurY <= ToY; CurY++){
209  if(CanPlaceAsset(CPosition(RightX, CurY), placeasset->Size(), placeasset)){
210  CPosition TempPosition(RightX, CurY);
211  CurDistance = TempPosition.DistanceSquared(nexttiletarget);
212  if((-1 == BestDistance)||(CurDistance < BestDistance)){
213  BestDistance = CurDistance;
214  BestPosition = TempPosition;
215  }
216  }
217  }
218  }
219  else{
220  Skipped++;
221  }
222  if(Height() > BottomY){
223  int ToX = MAX(LeftX, 0);
224  for(int CurX = MIN(RightX, Width() - 1); CurX >= ToX; CurX--){
225  if(CanPlaceAsset(CPosition(CurX, BottomY), placeasset->Size(), placeasset)){
226  CPosition TempPosition(CurX, BottomY);
227  CurDistance = TempPosition.DistanceSquared(nexttiletarget);
228  if((-1 == BestDistance)||(CurDistance < BestDistance)){
229  BestDistance = CurDistance;
230  BestPosition = TempPosition;
231  }
232  }
233  }
234  }
235  else{
236  Skipped++;
237  }
238  if(0 <= LeftX){
239  int ToY = MAX(TopY, 0);
240  for(int CurY = MIN(BottomY, Height() - 1); CurY >= ToY; CurY--){
241  if(CanPlaceAsset(CPosition(LeftX, CurY), placeasset->Size(), placeasset)){
242  CPosition TempPosition(LeftX, CurY);
243  CurDistance = TempPosition.DistanceSquared(nexttiletarget);
244  if((-1 == BestDistance)||(CurDistance < BestDistance)){
245  BestDistance = CurDistance;
246  BestPosition = TempPosition;
247  }
248  }
249  }
250  }

252  Skipped++;
253  }

255  break;
256  }

263  RightX++;

285 bool CAssetDecoratedMap::LoadMap(std::shared_ptr< CDataSource > source){
286  CLineDataSource LineSource(source);
287  std::string TempString;
288  std::vector< std::string > Tokens;
289  SResourceInitialization TempResourceInit;
290  SAssetInitialization TempAssetInit;
291  int ResourceCount, AssetCount;
292  bool ReturnStatus = false;
293 
294  if(!CTerrainMap::LoadMap(source)){
295  return false;
296  }
297  try{
298  if(!LineSource.Read(TempString)){
299  PrintError("Failed to read map resource count.\n");
300  goto LoadMapExit;
301  }
302  ResourceCount = std::stoi(TempString);
303  DResourceInitializationList.clear();
304  for(int Index = 0; Index <= ResourceCount; Index++){
305  if(!LineSource.Read(TempString)){
306  PrintError("Failed to read map resource %d.\n", Index);
307  goto LoadMapExit;
308  }
309  CTokenizer::Tokenize(Tokens, TempString);
310  if(3 > Tokens.size()){
311  PrintError("Too few tokens for resource %d.\n", Index);
312  goto LoadMapExit;
313  }
314  TempResourceInit.DColor = (EPlayerColor)std::stoi(Tokens[0]);
315  if((0 == Index)&&(pcNone != TempResourceInit.DColor)){
316  PrintError("Expected first resource to be for color None.\n");
317  goto LoadMapExit;
318  }
319  TempResourceInit.DGold = std::stoi(Tokens[1]);
320  TempResourceInit.DLumber = std::stoi(Tokens[2]);
321 
322  DResourceInitializationList.push_back(TempResourceInit);
323  }
324 
325 
326  if(!LineSource.Read(TempString)){
327  PrintError("Failed to read map asset count.\n");
328  goto LoadMapExit;
329  }
330  AssetCount = std::stoi(TempString);
331  DAssetInitializationList.clear();
332  for(int Index = 0; Index < AssetCount; Index++){
333  if(!LineSource.Read(TempString)){
334  PrintError("Failed to read map asset %d.\n", Index);
335  goto LoadMapExit;
336  }
337  CTokenizer::Tokenize(Tokens, TempString);
338  if(4 > Tokens.size()){
339  PrintError("Too few tokens for asset %d.\n", Index);
340  goto LoadMapExit;
341  }
342  TempAssetInit.DType = Tokens[0];
343  TempAssetInit.DColor = (EPlayerColor)std::stoi(Tokens[1]);
344  TempAssetInit.DTilePosition.X(std::stoi(Tokens[2]));
345  TempAssetInit.DTilePosition.Y(std::stoi(Tokens[3]));
346 
347  if((0 > TempAssetInit.DTilePosition.X())||(0 > TempAssetInit.DTilePosition.Y())){
348  PrintError("Invalid resource position %d (%d, %d).\n", Index, TempAssetInit.DTilePosition.X(), TempAssetInit.DTilePosition.Y());
349  goto LoadMapExit;
350  }
351  if((Width() <= TempAssetInit.DTilePosition.X())||(Height() <= TempAssetInit.DTilePosition.Y())){
352  PrintError("Invalid resource position %d (%d, %d).\n", Index, TempAssetInit.DTilePosition.X(), TempAssetInit.DTilePosition.Y());
353  goto LoadMapExit;
354  }
355  DAssetInitializationList.push_back(TempAssetInit);
356  }
357  ReturnStatus = true;
358  }
359  catch(std::exception &E){
360  PrintError("%s\n",E.what());
361  }
362 
363 LoadMapExit:
364  return ReturnStatus;
365 
366 }

400 bool CAssetDecoratedMap::UpdateMap(const CVisibilityMap &vismap, const CAssetDecoratedMap &resmap){
401  auto Iterator = DAssets.begin();
402 
403  if(DMap.size() != resmap.DMap.size()){
404  DMap.resize(resmap.DMap.size());
405  for(auto &Row : DMap){
406  Row.resize(resmap.DMap[0].size());
407  for(auto &Cell : Row){
408  Cell = ttNone;
409  }
410  }
411  }
412  while(Iterator != DAssets.end()){
413  CPosition CurPosition = (*Iterator)->TilePosition();
414  int AssetSize = (*Iterator)->Size();
415  bool RemoveAsset = false;
416  if((*Iterator)->Speed()||(aaDecay == (*Iterator)->Action())||(aaAttack == (*Iterator)->Action())){ // Remove all movable units
417  Iterator = DAssets.erase(Iterator);
418  continue;
419  }
420  for(int YOff = 0; YOff < AssetSize; YOff++){
421  int YPos = CurPosition.Y() + YOff;
422  for(int XOff = 0; XOff < AssetSize; XOff++){
423  int XPos = CurPosition.X() + XOff;
424 
425  CVisibilityMap::ETileVisibility VisType = vismap.TileType(XPos, YPos);
426  if((CVisibilityMap::tvPartial == VisType)||(CVisibilityMap::tvPartialPartial == VisType)||(CVisibilityMap::tvVisible == VisType)){ // Remove visible so they can be updated
427  RemoveAsset = atNone != (*Iterator)->Type();
428  break;
429  }
430  }
431  if(RemoveAsset){
432  break;
433  }
434  }
435  if(RemoveAsset){
436  Iterator = DAssets.erase(Iterator);
437  continue;
438  }
439  Iterator++;
440  }
441  for(int YPos = 0; YPos < DMap.size(); YPos++){
442  for(int XPos = 0; XPos < DMap[YPos].size(); XPos++){
443  CVisibilityMap::ETileVisibility VisType = vismap.TileType(XPos-1, YPos-1);
444  if((CVisibilityMap::tvPartial == VisType)||(CVisibilityMap::tvPartialPartial == VisType)||(CVisibilityMap::tvVisible == VisType)){
445  DMap[YPos][XPos] = resmap.DMap[YPos][XPos];
446  }
447  }
448  }
449  for(auto &Asset : resmap.DAssets){
450  CPosition CurPosition = Asset->TilePosition();
451  int AssetSize = Asset->Size();
452  bool AddAsset = false;
453 
454  for(int YOff = 0; YOff < AssetSize; YOff++){
455  int YPos = CurPosition.Y() + YOff;
456  for(int XOff = 0; XOff < AssetSize; XOff++){
457  int XPos = CurPosition.X() + XOff;
458 
459  CVisibilityMap::ETileVisibility VisType = vismap.TileType(XPos, YPos);
460  if((CVisibilityMap::tvPartial == VisType)||(CVisibilityMap::tvPartialPartial == VisType)||(CVisibilityMap::tvVisible == VisType)){ // Add visible resources
461  AddAsset = true;
462  break;
463  }
464  }
465  if(AddAsset){
466  DAssets.push_back(Asset);
467  break;
468  }
469  }
470  }
471 
472  return true;
473 }
