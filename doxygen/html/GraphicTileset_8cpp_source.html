47 int CGraphicTileset::TileCount(int count){
48  GdkPixbuf *TempPixbufTileset;
49  guchar *Pixels;
50  guint PixelLength;
51 
52  if(0 > count){
53  return false;
54  }
55  if(!DTileWidth || !DTileHeight){
56  return false;
57  }
58  if(count < DTileCount){
59  auto Iterator = DMapping.begin();
60  DTileCount = count;
61 
62  while(DMapping.end() != Iterator){
63  if(Iterator->second >= DTileCount){
64  Iterator = DMapping.erase(Iterator);
65  }
66  else{
67  Iterator++;
68  }
69  }
70  return true;
71  }
72  TempPixbufTileset = gdk_pixbuf_new(GDK_COLORSPACE_RGB, TRUE, 8, DTileWidth, count * DTileHeight);
73  if(nullptr == TempPixbufTileset){
74  return false;
75  }
76  Pixels = gdk_pixbuf_get_pixels_with_length(TempPixbufTileset, &PixelLength);
77  memset(Pixels, 0, PixelLength);
78  if(nullptr != DPixbufTileset){
79  int NumberChannels;
80  NumberChannels = gdk_pixbuf_get_n_channels(DPixbufTileset);
81 
82  memcpy(Pixels, gdk_pixbuf_get_pixels(DPixbufTileset), NumberChannels * DTileHeight * DTileWidth * DTileCount);
83  g_object_unref(G_OBJECT(DPixbufTileset));
84  }
85  DPixbufTileset = TempPixbufTileset;
86  DTileCount = count;
87  return true;
88 }

171 bool CGraphicTileset::LoadTileset(std::shared_ptr< CDataSource > source){
172  CLineDataSource LineSource(source);
173  std::string PNGPath, TempString;
174  std::vector< std::string > Tokens;
175  bool ReturnStatus = false;
176 
177  if(nullptr == source){
178  return false;
179  }
180  if(!LineSource.Read(PNGPath)){
181  PrintError("Failed to get path.\n");
182  goto LoadTilesetExit;
183  }
184  DPixbufTileset = CGraphicLoader::LoadPixbuf(source->Container()->DataSource(PNGPath));
185  if(nullptr == DPixbufTileset){
186  PrintError("Failed to load file %s.\n", PNGPath.c_str());
187  goto LoadTilesetExit;
188  }
189  DTileWidth = gdk_pixbuf_get_width(DPixbufTileset);
190  DTileHeight = gdk_pixbuf_get_height(DPixbufTileset);
191  if(gdk_pixbuf_get_has_alpha(DPixbufTileset)){
192  PrintDebug(DEBUG_LOW,"File %s has alpha channel\n", PNGPath.c_str());
193  }
194 
195  if(!LineSource.Read(TempString)){
196  goto LoadTilesetExit;
197  }
198 
199  DTileCount = atoi(TempString.c_str());
200  DTileHeight /= DTileCount;
201  for(int Index = 0; Index < DTileCount; Index++){
202  if(!LineSource.Read(TempString)){
203  goto LoadTilesetExit;
204  }
205  DMapping[TempString] = Index;
206  }
207  DTileHalfWidth = DTileWidth / 2;
208  DTileHalfHeight = DTileHeight / 2;
209  for(int Index = 0; Index < DTileCount; Index++){
210  GdkBitmap *NewClipMask;
211 
212  NewClipMask = gdk_pixmap_new(NULL, DTileWidth, DTileHeight, 1);
213  gdk_pixbuf_render_threshold_alpha(DPixbufTileset, NewClipMask, 0, Index * DTileHeight, 0, 0, DTileWidth, DTileHeight, 1);
214 
215  DClippingMasks.push_back(NewClipMask);
216  }
217  DPixelColors.clear();
218  DPixelMapping.clear();
219  PrintDebug(DEBUG_LOW, "Tile Dims %d x %d\n", DTileWidth, DTileHeight);
220  if(!LineSource.Read(TempString)){
221  ReturnStatus = true;
222  goto LoadTilesetExit;
223  }
224 
225  try{
226  DPixelCount = std::stoi(TempString);
227  for(int Index = 0; Index < DPixelCount; Index++){
228  GdkColor TempColor;
229  std::size_t NextPos;
230  if(!LineSource.Read(TempString)){
231  PrintError("Failed to read %d pixel color.\n", Index);
232  goto LoadTilesetExit;
233  }
234  CTokenizer::Tokenize(Tokens, TempString);
235  if(Tokens.size() != 2){
236  PrintError("Failed pixel color %d does not have 2 tokens.\n", Index);
237  goto LoadTilesetExit;
238  }
239  TempColor.pixel = std::stoi(Tokens[1], &NextPos, 16);
240  TempColor.red = (TempColor.pixel>>8) & 0xFF00;
241  TempColor.green = TempColor.pixel & 0xFF00;
242  TempColor.blue = (TempColor.pixel<<8) & 0xFF00;
243  DPixelMapping[Tokens[0]] = DPixelColors.size();
244  DPixelColors.push_back(TempColor);
245  }
246  }
247  catch(std::exception &E){
248  PrintError("%s\n",E.what());
249  }
250  ReturnStatus = true;
251 LoadTilesetExit:
252  return ReturnStatus;
253 }
