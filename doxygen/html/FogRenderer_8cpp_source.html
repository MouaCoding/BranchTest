19 static int HammingDistance(int v1, int v2){
20  int Delta = v1 ^ v2;
21  int Distance = 0;
22 
23  while(Delta){
24  if(Delta & 0x01){
25  Distance++;
26  }
27  Delta >>= 1;
28  }
29  return Distance;
30 }

184 void CFogRenderer::DrawMap(GdkDrawable *drawable, const SRectangle &rect){
185  GdkGC *TempGC = gdk_gc_new(drawable);
186  int TileWidth, TileHeight;
187  static std::vector< bool > UnknownFog, UnknownBlack;
188 
189  if((UnknownFog.size() < 0x100)||(UnknownBlack.size() < 0x100)){
190  UnknownFog.resize(0x100);
191  UnknownBlack.resize(0x100);
192  }
193 
194  TileWidth = DTileset->TileWidth();
195  TileHeight = DTileset->TileHeight();
196 
197  for(int YIndex = rect.DYPosition / TileHeight, YPos = -(rect.DYPosition % TileHeight); YPos < rect.DHeight; YIndex++, YPos += TileHeight){
198  for(int XIndex = rect.DXPosition / TileWidth, XPos = -(rect.DXPosition % TileWidth); XPos < rect.DWidth; XIndex++, XPos += TileWidth){
199  CVisibilityMap::ETileVisibility TileType = DMap->TileType(XIndex, YIndex);
200 
201  if(CVisibilityMap::tvNone == TileType){
202  DTileset->DrawTile(drawable, TempGC, XPos, YPos, DNoneIndex);
203  continue;
204  }
205  else if(CVisibilityMap::tvVisible == TileType){
206  continue;
207  }
208  if((CVisibilityMap::tvSeen == TileType)||(CVisibilityMap::tvSeenPartial == TileType)){
209  DTileset->DrawTile(drawable, TempGC, XPos, YPos, DSeenIndex);
210  }
211  if((CVisibilityMap::tvPartialPartial == TileType)||(CVisibilityMap::tvPartial == TileType)){
212  int VisibilityIndex = 0, VisibilityMask = 0x1;
213 
214  for(int YOff = -1; YOff < 2; YOff++){
215  for(int XOff = -1; XOff < 2; XOff++){
216  if(YOff || XOff){
217  CVisibilityMap::ETileVisibility VisTile = DMap->TileType(XIndex + XOff, YIndex + YOff);
218 
219  if(CVisibilityMap::tvVisible == VisTile){
220  VisibilityIndex |= VisibilityMask;
221  }
222  VisibilityMask <<= 1;
223  }
224  }
225  }
226  if(-1 == DFogIndices[VisibilityIndex]){
227  if(!UnknownFog[VisibilityIndex]){
228  printf("Unknown fog 0x%02X @ (%d, %d)\n",VisibilityIndex, XIndex, YIndex);
229  UnknownFog[VisibilityIndex] = true;
230  }
231  }
232  DTileset->DrawTile(drawable, TempGC, XPos, YPos, DFogIndices[VisibilityIndex]);
233  }
234 
235  if((CVisibilityMap::tvPartialPartial == TileType)||(CVisibilityMap::tvSeenPartial == TileType)){
236  int VisibilityIndex = 0, VisibilityMask = 0x1;
237 
238  for(int YOff = -1; YOff < 2; YOff++){
239  for(int XOff = -1; XOff < 2; XOff++){
240  if(YOff || XOff){
241  CVisibilityMap::ETileVisibility VisTile = DMap->TileType(XIndex + XOff, YIndex + YOff);
242 
243  if((CVisibilityMap::tvVisible == VisTile)||(CVisibilityMap::tvPartial == VisTile)||(CVisibilityMap::tvSeen == VisTile)){
244  VisibilityIndex |= VisibilityMask;
245  }
246  VisibilityMask <<= 1;
247  }
248  }
249  }
250  if(-1 == DBlackIndices[VisibilityIndex]){
251  if(!UnknownBlack[VisibilityIndex]){
252  printf("Unknown black 0x%02X @ (%d, %d)\n",VisibilityIndex, XIndex, YIndex);
253  UnknownBlack[VisibilityIndex] = true;
254  }
255  }
256  DTileset->DrawTile(drawable, TempGC, XPos, YPos, DBlackIndices[VisibilityIndex]);
257  }
258  }
259  }
260  g_object_unref(TempGC);
261 }
