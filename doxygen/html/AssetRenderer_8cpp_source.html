297 void CAssetRenderer::DrawAssets(GdkDrawable *drawable, GdkDrawable *typedrawable, const SRectangle &rect){
298  GdkGC *TempGC = gdk_gc_new(drawable);
299  GdkGC *TempTypeGC = gdk_gc_new(typedrawable);
300  int ScreenRightX = rect.DXPosition + rect.DWidth - 1;
301  int ScreenBottomY = rect.DYPosition + rect.DHeight - 1;
302  std::list< SAssetRenderData > FinalRenderList;
303 
304  for(auto &AssetIterator : DPlayerMap->Assets()){
305  SAssetRenderData TempRenderData;
306  TempRenderData.DType = AssetIterator->Type();
307  if(atNone == TempRenderData.DType){
308  continue;
309  }
310  if((0 <= TempRenderData.DType)&&(TempRenderData.DType < DTilesets.size())){
311  CPixelType PixelType(*AssetIterator);
312  int RightX;
313  TempRenderData.DX = AssetIterator->PositionX() + (AssetIterator->Size() - 1) * CPosition::HalfTileWidth() - DTilesets[TempRenderData.DType]->TileHalfWidth();
314  TempRenderData.DY = AssetIterator->PositionY() + (AssetIterator->Size() - 1) * CPosition::HalfTileHeight() - DTilesets[TempRenderData.DType]->TileHalfHeight();
315  TempRenderData.DPixelColor = PixelType.ToPixelColor();
316  //printf("%d @ (%d, %d)\n",TempRenderData.DType, TempRenderData.DX , TempRenderData.DY);
317  RightX = TempRenderData.DX + DTilesets[TempRenderData.DType]->TileWidth() - 1;
318  TempRenderData.DBottomY = TempRenderData.DY + DTilesets[TempRenderData.DType]->TileHeight() - 1;
319  bool OnScreen = true;
320  if((RightX < rect.DXPosition)||(TempRenderData.DX > ScreenRightX)){
321  OnScreen = false;
322  }
323  else if((TempRenderData.DBottomY < rect.DYPosition)||(TempRenderData.DY > ScreenBottomY)){
324  OnScreen = false;
325  }
326  TempRenderData.DX -= rect.DXPosition;
327  TempRenderData.DY -= rect.DYPosition;
328  TempRenderData.DColorIndex = AssetIterator->Color() ? AssetIterator->Color() - 1 : AssetIterator->Color();
329  TempRenderData.DTileIndex = -1;
330  if(OnScreen){
331  int ActionSteps, CurrentStep, TileIndex;
332  switch(AssetIterator->Action()){
333  case aaBuild: ActionSteps = DBuildIndices[TempRenderData.DType].size();
334  ActionSteps /= dMax;
335  if(ActionSteps){
336  TileIndex = AssetIterator->Direction() * ActionSteps + ((AssetIterator->Step() / DAnimationDownsample)% ActionSteps);
337  TempRenderData.DTileIndex = DBuildIndices[TempRenderData.DType][TileIndex];
338  }
339  break;
340  case aaConstruct: ActionSteps = DConstructIndices[TempRenderData.DType].size();
341  if(ActionSteps){
342  int TotalSteps = AssetIterator->BuildTime() * CPlayerAsset::UpdateFrequency();
343  int CurrentStep = AssetIterator->Step() * ActionSteps / TotalSteps;
344  if(CurrentStep == DConstructIndices[TempRenderData.DType].size()){
345  CurrentStep--;
346  }
347  TempRenderData.DTileIndex = DConstructIndices[TempRenderData.DType][CurrentStep];
348  }
349  break;
350  case aaWalk: if(AssetIterator->Lumber()){
351  ActionSteps = DCarryLumberIndices[TempRenderData.DType].size();
352  ActionSteps /= dMax;
353  TileIndex = AssetIterator->Direction() * ActionSteps + ((AssetIterator->Step() / DAnimationDownsample)% ActionSteps);
354  TempRenderData.DTileIndex = DCarryLumberIndices[TempRenderData.DType][TileIndex];
355  }
356  else if(AssetIterator->Gold()){
357  ActionSteps = DCarryGoldIndices[TempRenderData.DType].size();
358  ActionSteps /= dMax;
359  TileIndex = AssetIterator->Direction() * ActionSteps + ((AssetIterator->Step() / DAnimationDownsample)% ActionSteps);
360  TempRenderData.DTileIndex = DCarryGoldIndices[TempRenderData.DType][TileIndex];
361  }
362  else{
363  ActionSteps = DWalkIndices[TempRenderData.DType].size();
364  ActionSteps /= dMax;
365  TileIndex = AssetIterator->Direction() * ActionSteps + ((AssetIterator->Step() / DAnimationDownsample)% ActionSteps);
366  TempRenderData.DTileIndex = DWalkIndices[TempRenderData.DType][TileIndex];
367  }
368  break;
369  case aaAttack: CurrentStep = AssetIterator->Step() % (AssetIterator->AttackSteps() + AssetIterator->ReloadSteps());
370  if(CurrentStep < AssetIterator->AttackSteps()){
371  ActionSteps = DAttackIndices[TempRenderData.DType].size();
372  ActionSteps /= dMax;
373  TileIndex = AssetIterator->Direction() * ActionSteps + (CurrentStep * ActionSteps / AssetIterator->AttackSteps());
374  TempRenderData.DTileIndex = DAttackIndices[TempRenderData.DType][TileIndex];
375  }
376  else{
377  TempRenderData.DTileIndex = DNoneIndices[TempRenderData.DType][AssetIterator->Direction()];
378  }
379  break;
380  case aaRepair:
381  case aaHarvestLumber: ActionSteps = DAttackIndices[TempRenderData.DType].size();
382  ActionSteps /= dMax;
383  TileIndex = AssetIterator->Direction() * ActionSteps + ((AssetIterator->Step() / DAnimationDownsample)% ActionSteps);
384  TempRenderData.DTileIndex = DAttackIndices[TempRenderData.DType][TileIndex];
385  break;
386  case aaMineGold: break;
387  case aaStandGround:
388  case aaNone: TempRenderData.DTileIndex = DNoneIndices[TempRenderData.DType][AssetIterator->Direction()];
389  if(AssetIterator->Speed()){
390  if(AssetIterator->Lumber()){
391  ActionSteps = DCarryLumberIndices[TempRenderData.DType].size();
392  ActionSteps /= dMax;
393  TempRenderData.DTileIndex = DCarryLumberIndices[TempRenderData.DType][AssetIterator->Direction() * ActionSteps];
394  }
395  else if(AssetIterator->Gold()){
396  ActionSteps = DCarryGoldIndices[TempRenderData.DType].size();
397  ActionSteps /= dMax;
398  TempRenderData.DTileIndex = DCarryGoldIndices[TempRenderData.DType][AssetIterator->Direction() * ActionSteps];
399  }
400  }
401  break;
402  case aaCapability: if(AssetIterator->Speed()){
403  if((actPatrol == AssetIterator->CurrentCommand().DCapability)||(actStandGround == AssetIterator->CurrentCommand().DCapability)){
404  TempRenderData.DTileIndex = DNoneIndices[TempRenderData.DType][AssetIterator->Direction()];
405  }
406  }
407  else{
408  // Buildings
409  TempRenderData.DTileIndex = DNoneIndices[TempRenderData.DType][AssetIterator->Direction()];
410  }
411  break;
412  case aaDeath: ActionSteps = DDeathIndices[TempRenderData.DType].size();
413  if(AssetIterator->Speed()){
414  ActionSteps /= dMax;
415  if(ActionSteps){
416  CurrentStep = AssetIterator->Step() / DAnimationDownsample;
417  if(CurrentStep >= ActionSteps){
418  CurrentStep = ActionSteps - 1;
419  }
420  TempRenderData.DTileIndex = DDeathIndices[TempRenderData.DType][AssetIterator->Direction() * ActionSteps + CurrentStep];
421  }
422  }
423  else{
424  if(AssetIterator->Step() < DBuildingDeathTileset->TileCount()){
425  TempRenderData.DTileIndex = DTilesets[TempRenderData.DType]->TileCount() + AssetIterator->Step();
426  TempRenderData.DX += DTilesets[TempRenderData.DType]->TileHalfWidth() - DBuildingDeathTileset->TileHalfWidth();
427  TempRenderData.DY += DTilesets[TempRenderData.DType]->TileHalfHeight() - DBuildingDeathTileset->TileHalfHeight();
428  }
429  }
430  default: break;
431  }
432  if(0 <= TempRenderData.DTileIndex){
433  FinalRenderList.push_back(TempRenderData);
434  }
435  }
436  }
437  }
438  FinalRenderList.sort(CompareRenderData);
439  for(auto &RenderIterator : FinalRenderList){
440  if(RenderIterator.DTileIndex < DTilesets[RenderIterator.DType]->TileCount()){
441  DTilesets[RenderIterator.DType]->DrawTile(drawable, TempGC, RenderIterator.DX, RenderIterator.DY, RenderIterator.DTileIndex, RenderIterator.DColorIndex);
442  DTilesets[RenderIterator.DType]->DrawClipped(typedrawable, TempTypeGC, RenderIterator.DX, RenderIterator.DY, RenderIterator.DTileIndex, RenderIterator.DPixelColor);
443  }
444  else{
445  DBuildingDeathTileset->DrawTile(drawable, TempGC, RenderIterator.DX, RenderIterator.DY, RenderIterator.DTileIndex);
446  }
447  }
448  g_object_unref(TempTypeGC);
449  g_object_unref(TempGC);
450 }
