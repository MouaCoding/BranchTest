110 int CSoundLibraryMixer::Timestep(void *out, unsigned long frames, const PaStreamCallbackTimeInfo* timeinfo, PaStreamCallbackFlags status){
111  float *DataPtr = (float *)out;
112 
113  memset(DataPtr, 0, sizeof(float) * frames * 2);
114  //pthread_mutex_lock(&DMutex);
115  DMutex.lock();
116  for(std::list< SClipStatus >::iterator ClipIterator = DClipsInProgress.begin(); ClipIterator != DClipsInProgress.end(); ){
117  bool Advance = true;
118 
119  DSoundClips[ClipIterator->DIndex].MixStereoClip(DataPtr, ClipIterator->DOffset, frames, ClipIterator->DVolume, ClipIterator->DRightBias);
120 
121  ClipIterator->DOffset += frames;
122  if(ClipIterator->DOffset >= DSoundClips[ClipIterator->DIndex].TotalFrames()){
123  DFreeClipIDs.push_back(ClipIterator->DIdentification);
124  ClipIterator = DClipsInProgress.erase(ClipIterator);
125  Advance = false;
126  }
127  if(Advance){
128  ClipIterator++;
129  }
130  }
131 
132  for(std::list< SToneStatus >::iterator ToneIterator = DTonesInProgress.begin(); ToneIterator != DTonesInProgress.end(); ){
133  bool Advance = true;
134  float *CurOutPtr = DataPtr;
135 
136  for(int Frame = 0; Frame < frames; Frame++){
137  int SineIndex = ToneIterator->DCurrentStep;
138 
139  if(0.5 <= (ToneIterator->DCurrentStep - SineIndex)){
140  SineIndex++;
141  if(DSampleRate <= SineIndex){
142  SineIndex = 0;
143  }
144  }
145  *CurOutPtr++ += ToneIterator->DVolume * (1.0 - ToneIterator->DRightBias) * DSineWave[SineIndex];
146  *CurOutPtr++ += ToneIterator->DVolume * (1.0 + ToneIterator->DRightBias) * DSineWave[SineIndex];
147  ToneIterator->DCurrentStep += ToneIterator->DCurrentFrequency;
148  ToneIterator->DCurrentFrequency += ToneIterator->DFrequencyDecay;
149  ToneIterator->DVolume += ToneIterator->DVolumeDecay;
150  ToneIterator->DRightBias += ToneIterator->DRightShift;
151  if(DSampleRate <= ToneIterator->DCurrentStep){
152  ToneIterator->DCurrentStep -= DSampleRate;
153  }
154  if(0.0 > ToneIterator->DVolume){
155  Advance = false;
156  break;
157  }
158  if(20.0 > ToneIterator->DCurrentFrequency){
159  Advance = false;
160  break;
161  }
162  if(-1.0 > ToneIterator->DRightBias){
163  ToneIterator->DRightBias = -1.0;
164  ToneIterator->DRightShift = 0.0;
165  }
166  if(1.0 < ToneIterator->DRightBias){
167  ToneIterator->DRightBias = -1.0;
168  ToneIterator->DRightShift = 0.0;
169  }
170  }
171  if(Advance){
172  ToneIterator++;
173  }
174  else{
175  DFreeToneIDs.push_back(ToneIterator->DIdentification);
176  ToneIterator = DTonesInProgress.erase(ToneIterator);
177  }
178  }
179 
180  //pthread_mutex_unlock(&DMutex);
181  DMutex.unlock();
182  frames *= 2;
183  for(int Frame = 0; Frame < frames; Frame++){
184  if(-1.0 > *DataPtr){
185  *DataPtr = -1.0;
186  }
187  if(1.0 < *DataPtr){
188  *DataPtr = 1.0;
189  }
190  DataPtr++;
191  }
192 
193  return paContinue;
194 }

196 bool CSoundLibraryMixer::LoadLibrary(std::shared_ptr< CDataSource > source){
197  CLineDataSource LineSource(source);
198  int TotalClips, TotalSongs;
199  bool ReturnStatus = false;
200  std::string TempString;
201  std::shared_ptr< CDataSource > SFInput;
202  std::ofstream SFOutput;
203  char Buffer[1024];
204  int BytesRead;
205 
206  if(!LineSource.Read(TempString)){
207  PrintError("Failed to read clip count.\n");
208  goto LoadLibraryExit;
209  }
210  try{
211  TotalClips = std::stoi(TempString);
212  }
213  catch(std::exception &E){
214  PrintError("%s\n",E.what());
215  goto LoadLibraryExit;
216  }
217 
218  DSoundClips.resize(TotalClips);
219  for(int Index = 0; Index < TotalClips; Index++){
220  if(!LineSource.Read(TempString)){
221  PrintError("Failed to read clip name %d.\n", Index);
222  goto LoadLibraryExit;
223  }
224  DMapping[TempString] = Index;
225  if(!LineSource.Read(TempString)){
226  PrintError("Failed to read clip path %d.\n", Index);
227  goto LoadLibraryExit;
228  }
229  if(!DSoundClips[Index].Load(source->Container()->DataSource(TempString))){
230  PrintError("Failed to load sound clip %d.\n", Index);
231  goto LoadLibraryExit;
232  }
233  }
234  DSampleRate = DSoundClips[0].SampleRate();
235  for(int Index = 1; Index < TotalClips; Index++){
236  if(DSoundClips[0].SampleRate() != DSoundClips[Index].SampleRate()){
237  PrintError("Incompatible sample rate %dHz of clip %d.\n", DSoundClips[Index].SampleRate(), Index);
238  goto LoadLibraryExit;
239  }
240  }
241  if(NULL != DSineWave){
242  delete [] DSineWave;
243  }
244  DSineWave = new float[DSampleRate];
245  for(int Index = 0; Index < DSampleRate; Index++){
246  DSineWave[Index] = sin( ((double)Index/(double)DSampleRate) * M_PI * 2.0 );
247  }
248  if(!LineSource.Read(TempString)){
249  PrintError("Failed to read sound font name!\n");
250  goto LoadLibraryExit;
251  }
252  if(!fluid_is_soundfont(TEMP_SFNAME)){
253  PrintDebug(DEBUG_LOW, "Copying sound font %s to %s\n",TempString.c_str(), TEMP_SFNAME);
254  SFInput = source->Container()->DataSource(TempString);
255  SFOutput.open(TEMP_SFNAME, std::ofstream::binary);
256  while(0 < (BytesRead = SFInput->Read(Buffer, sizeof(Buffer)))){
257  SFOutput.write(Buffer, BytesRead);
258  }
259  SFOutput.close();
260  PrintDebug(DEBUG_LOW, "Copying complete.\nChecking if valid sound font.\n");
261  }
262  if(!fluid_is_soundfont(TEMP_SFNAME)){
263  PrintError("%s not a sound font!\n",TempString.c_str());
264  goto LoadLibraryExit;
265  }
266  if(!LineSource.Read(TempString)){
267  PrintError("Failed to read line\n");
268  goto LoadLibraryExit;
269  }
270  try{
271  TotalSongs = std::stoi(TempString);
272  }
273  catch(std::exception &E){
274  PrintError("%s\n",E.what());
275  goto LoadLibraryExit;
276  }
277  DMusicFilenames.resize(TotalSongs);
278  for(int Index = 0; Index < TotalSongs; Index++){
279  auto TempData = std::make_shared< std::vector< char > > ();
280 
281  if(!LineSource.Read(TempString)){
282  PrintError("Failed to read song name %d\n", Index);
283  goto LoadLibraryExit;
284  }
285  DMusicMapping[TempString] = Index;
286  if(!LineSource.Read(TempString)){
287  PrintError("Failed to read song path %d\n", Index);
288  goto LoadLibraryExit;
289  }
290  auto MidiInput = source->Container()->DataSource(TempString);
291  while(0 < (BytesRead = MidiInput->Read(Buffer, sizeof(Buffer)))){
292  TempData->insert(TempData->end(), Buffer, Buffer + BytesRead);
293  }
294  DMusicData.push_back(TempData);
295  /*
296  if(!fluid_is_midifile(TempString.c_str())){
297  PrintError("%s is not a midi file!\n", TempString.c_str());
298  goto LoadLibraryExit;
299  }
300  */
301  DMusicFilenames[Index] = TempString;
302  }
303 
304  DFluidSettings = new_fluid_settings();
305  fluid_settings_setstr(DFluidSettings, "audio.driver", "portaudio");
306  DFluidSynthesizer = new_fluid_synth(DFluidSettings);
307  DFluidPlayer = new_fluid_player(DFluidSynthesizer);
308  DFluidAudioDriver = new_fluid_audio_driver(DFluidSettings, DFluidSynthesizer);
309  PrintDebug(DEBUG_LOW, "Loading sound font %s.\n", TEMP_SFNAME);
310  if(FLUID_FAILED == fluid_synth_sfload(DFluidSynthesizer, TEMP_SFNAME, 1)){
311  PrintError("Failed to load sound font %s\n", TEMP_SFNAME);
312  goto LoadLibraryExit;
313  }
314  PrintDebug(DEBUG_LOW, "Sound font %s loaded.\n", TEMP_SFNAME);
315  if(!DPortAudioInitialized){
316  PaError PortAudioResult;
317  PrintDebug(DEBUG_LOW, "Port Audio not initialized, try to reinitialize.\n");
318  PortAudioResult = Pa_Initialize();
319  DPortAudioInitialized = paNoError == PortAudioResult;
320  if(DPortAudioInitialized){
321  PrintDebug(DEBUG_LOW, "Port Audio initialized.\n");
322  }
323  else{
324  PrintDebug(DEBUG_LOW, "Port Audio failed to initialize with code %d.\n", PortAudioResult);
325  }
326  }
327  PrintDebug(DEBUG_LOW, "Opening stream at %dHz with %d frames per buffer\n",DSampleRate, FRAMES_PER_BUFFER);
328  if(paNoError != Pa_OpenDefaultStream(&DStream, 0, 2, paFloat32, DSampleRate, FRAMES_PER_BUFFER, TimestepCallback, this)){
329  PrintError("Failed to open default sound stream\n");
330  goto LoadLibraryExit;
331  }
332  PrintDebug(DEBUG_LOW, "Default sound stream opened.\n");
333  if(paNoError != Pa_StartStream(DStream)){
334  PrintError("Failed to start default sound stream\n");
335  goto LoadLibraryExit;
336  }
337  PrintDebug(DEBUG_LOW, "Sound stream started.\n");
338  return true;
339 
340 LoadLibraryExit:
341  DMusicFilenames.clear();
342  DSoundClips.clear();
343  DMapping.clear();
344 
345  return ReturnStatus;
346 }
