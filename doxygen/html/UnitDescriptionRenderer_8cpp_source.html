94 std::string CUnitDescriptionRenderer::AddAssetNameSpaces(const std::string &name){

121 int CUnitDescriptionRenderer::Selection(const CPosition &pos) const{
122  int HorizontalIcons, VerticalIcons;
123  int HorizontalGap, VerticalGap;
124  int SelectedIcon = -1;
125 
126  HorizontalIcons = DDisplayedWidth / DFullIconWidth;
127  VerticalIcons = DDisplayedHeight / DFullIconHeight;
128  HorizontalGap = (DDisplayedWidth - (HorizontalIcons * DFullIconWidth)) / (HorizontalIcons - 1);
129  VerticalGap = (DDisplayedHeight - (VerticalIcons * DFullIconHeight)) / (VerticalIcons - 1);
130  if(((pos.X() % (DFullIconWidth + HorizontalGap)) < DFullIconWidth)&&((pos.Y() % (DFullIconHeight + VerticalGap)) < DFullIconHeight)){
131  SelectedIcon = (pos.X() / (DFullIconWidth + HorizontalGap)) + HorizontalIcons * (pos.Y() / (DFullIconHeight + VerticalGap));
132  if(DDisplayedIcons <= SelectedIcon){
133  SelectedIcon = -1;
134  }
135  }
136 
137  return SelectedIcon;
138 }
