29 bool CGraphicRecolorMap::Load(std::shared_ptr< CDataSource > source){
30  GdkPixbuf *ColorPixbuf;
31  bool ReturnStatus = false;
32  int NumberChannels, RowStride;
33 
34  if(nullptr == source){
35  return false;
36  }
37 
38  ColorPixbuf = CGraphicLoader::LoadPixbuf(source);
39  if(nullptr == ColorPixbuf){
40  printf("Failed to load file colormap.\n");
41  return false;
42  }
43  NumberChannels = gdk_pixbuf_get_n_channels(ColorPixbuf);
44  RowStride = gdk_pixbuf_get_rowstride(ColorPixbuf);
45  if(GDK_COLORSPACE_RGB != gdk_pixbuf_get_colorspace(ColorPixbuf)){
46  printf("Colormap not RGB.\n");
47  goto LoadExit;
48  }
49  if(8 != gdk_pixbuf_get_bits_per_sample(ColorPixbuf)){
50  printf("Colormap not 8-bit samples.\n");
51  goto LoadExit;
52  }
53  DColors.resize(gdk_pixbuf_get_height(ColorPixbuf));
54  for(size_t Index = 0; Index < DColors.size(); Index++){
55  for(int XPos = 0; XPos < gdk_pixbuf_get_width(ColorPixbuf); XPos++){
56  guchar *Pixel = gdk_pixbuf_get_pixels(ColorPixbuf) + RowStride * Index + XPos * NumberChannels;
57  SRGBColor TempColor;
58 
59  TempColor.DRed = Pixel[0];
60  TempColor.DGreen = Pixel[1];
61  TempColor.DBlue = Pixel[2];
62  DColors[Index].push_back(TempColor);
63  }
64  }
65  ReturnStatus = true;
66 LoadExit:
67  g_object_unref(G_OBJECT(ColorPixbuf));
68  return ReturnStatus;
69 }
