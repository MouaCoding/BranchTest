211 bool CAIPlayer::BuildBuilding(SPlayerCommandRequest &command, EAssetType buildingtype, EAssetType neartype){
212  std::shared_ptr< CPlayerAsset > BuilderAsset;
213  std::shared_ptr< CPlayerAsset > TownHallAsset;
214  std::shared_ptr< CPlayerAsset > NearAsset;
215  EAssetCapabilityType BuildAction;
216  bool AssetIsIdle = false;
217 
218  switch(buildingtype){
219  case atBarracks: BuildAction = actBuildBarracks;
220  break;
221  case atLumberMill: BuildAction = actBuildLumberMill;
222  break;
223  case atBlacksmith: BuildAction = actBuildBlacksmith;
224  break;
225  default: BuildAction = actBuildFarm;
226  break;
227  }
228 
229  for(auto WeakAsset : DPlayerData->Assets()){
230  if(auto Asset = WeakAsset.lock()){
231  if(Asset->HasCapability(BuildAction) && Asset->Interruptible()){
232  if(!BuilderAsset || (!AssetIsIdle && (aaNone == Asset->Action()))){
233  BuilderAsset = Asset;
234  AssetIsIdle = aaNone == Asset->Action();
235  }
236  }
237  if(Asset->HasCapability(actBuildPeasant)){
238  TownHallAsset = Asset;
239  }
240  if(Asset->HasActiveCapability(BuildAction)){
241  return false;
242  }
243  if((neartype == Asset->Type())&&(aaConstruct != Asset->Action())){
244  NearAsset = Asset;
245  }
246  if(buildingtype == Asset->Type()){
247  if(aaConstruct == Asset->Action()){
248  return false;
249  }
250  }
251  }
252  }
253  if((buildingtype != neartype) && !NearAsset){
254  return false;
255  }
256  if(BuilderAsset){
257  auto PlayerCapability = CPlayerCapability::FindCapability(BuildAction);
258  CPosition SourcePosition = TownHallAsset->TilePosition();
259  CPosition MapCenter(DPlayerData->PlayerMap()->Width()/2, DPlayerData->PlayerMap()->Height()/2);
260 
261 
262  if(NearAsset){
263  SourcePosition = NearAsset->TilePosition();
264  }
265  if(MapCenter.X() < SourcePosition.X()){
266  SourcePosition.DecrementX(TownHallAsset->Size()/2);
267  }
268  else if(MapCenter.X() > SourcePosition.X()){
269  SourcePosition.IncrementX(TownHallAsset->Size()/2);
270  }
271  if(MapCenter.Y() < SourcePosition.Y()){
272  SourcePosition.DecrementY(TownHallAsset->Size()/2);
273  }
274  else if(MapCenter.Y() > SourcePosition.Y()){
275  SourcePosition.IncrementY(TownHallAsset->Size()/2);
276  }
277 
278  CPosition Placement = DPlayerData->FindBestAssetPlacement(SourcePosition, BuilderAsset, buildingtype, 1);
279  if(0 > Placement.X()){
280  return SearchMap(command);
281  }
282  if(PlayerCapability){
283  if(PlayerCapability->CanInitiate(BuilderAsset, DPlayerData)){
284  if(0 <= Placement.X()){
285  command.DAction = BuildAction;
286  command.DActors.push_back(BuilderAsset);
287  command.DTargetLocation.SetFromTile(Placement);
288  return true;
289  }
290  }
291  }
292  }
293 
294  return false;
295 }

366  command.DTargetType = atGoldMine;
367  command.DTargetLocation = GoldMineAsset->Position();

373  auto PlayerCapability = CPlayerCapability::FindCapability(actBuildPeasant);

473 void CAIPlayer::CalculateCommand(SPlayerCommandRequest &command){
474  command.DAction = actNone;
475  command.DActors.clear();
476  command.DTargetColor = pcNone;
477  command.DTargetType = atNone;
478  if((DCycle % DDownSample) == 0){
479  // Do decision
480 
481  if(0 == DPlayerData->FoundAssetCount(atGoldMine)){
482  // Search for gold mine
483  SearchMap(command);
484  }
485  else if((0 == DPlayerData->PlayerAssetCount(atTownHall))&&(0 == DPlayerData->PlayerAssetCount(atKeep))&&(0 == DPlayerData->PlayerAssetCount(atCastle))){
486  BuildTownHall(command);
487  }
488  else if(5 > DPlayerData->PlayerAssetCount(atPeasant)){
489  ActivatePeasants(command, true);
490  }
491  else if(12 > DPlayerData->VisibilityMap()->SeenPercent(100)){
492  SearchMap(command);
493  }
494  else{
495  bool CompletedAction = false;
496  int BarracksCount = 0;
497  int FootmanCount = DPlayerData->PlayerAssetCount(atFootman);
498  int ArcherCount = DPlayerData->PlayerAssetCount(atArcher)+DPlayerData->PlayerAssetCount(atRanger);
499 
500  if(!CompletedAction && (DPlayerData->FoodConsumption() >= DPlayerData->FoodProduction())){
501  CompletedAction = BuildBuilding(command, atFarm, atFarm);
502  }
503  if(!CompletedAction){
504  CompletedAction = ActivatePeasants(command, false);
505  }
506  if(!CompletedAction && (0 == (BarracksCount = DPlayerData->PlayerAssetCount(atBarracks)))){
507  CompletedAction = BuildBuilding(command, atBarracks, atFarm);
508  }
509  if(!CompletedAction && (5 > FootmanCount)){
510  CompletedAction = TrainFootman(command);
511  }
512  if(!CompletedAction && (0 == DPlayerData->PlayerAssetCount(atLumberMill))){
513  CompletedAction = BuildBuilding(command, atLumberMill, atBarracks);
514  }
515  if(!CompletedAction && (5 > ArcherCount)){
516  CompletedAction = TrainArcher(command);
517  }
518  if(!CompletedAction && DPlayerData->PlayerAssetCount(atFootman)){
519  CompletedAction = FindEnemies(command);
520  }
521  if(!CompletedAction){
522  CompletedAction = ActivateFighters(command);
523  }
524  if(!CompletedAction && ((5 <= FootmanCount) && (5 <= ArcherCount))){
525  CompletedAction = AttackEnemies(command);
526  }
527  }
528  }
529  DCycle++;
530 }
