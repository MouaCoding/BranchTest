315 bool CPlayerCapabilityBuildRanger::CActivatedCapability::IncrementStep(){
316  if(atBarracks == DActor->Type()){
317  int AddHitPoints = (DTarget->MaxHitPoints() * (DCurrentStep + 1) / DTotalSteps) - (DTarget->MaxHitPoints() * DCurrentStep / DTotalSteps);
318 
319  DTarget->IncrementHitPoints(AddHitPoints);
320  if(DTarget->HitPoints() > DTarget->MaxHitPoints()){
321  DTarget->HitPoints(DTarget->MaxHitPoints());
322  }
323  }
324 
325  DCurrentStep++;
326  DActor->IncrementStep();
327  if(DCurrentStep >= DTotalSteps){
328  SGameEvent TempEvent;
329 
330  if(atLumberMill == DActor->Type()){
331  auto BarracksIterator = DPlayerData->AssetTypes()->find("Barracks");
332  auto RangerIterator = DPlayerData->AssetTypes()->find("Ranger");
333  auto LumberMillIterator = DPlayerData->AssetTypes()->find("LumberMill");
334 
335  TempEvent.DType = etWorkComplete;
336  TempEvent.DAsset = DActor;
337 
338  BarracksIterator->second->AddCapability(actBuildRanger);
339  BarracksIterator->second->RemoveCapability(actBuildArcher);
340  LumberMillIterator->second->AddCapability(actLongbow);
341  LumberMillIterator->second->AddCapability(actRangerScouting);
342  LumberMillIterator->second->AddCapability(actMarksmanship);
343  // Upgrade all Archers
344  for(auto WeakAsset : DPlayerData->Assets()){
345  if(auto Asset = WeakAsset.lock()){
346  if(atArcher == Asset->Type()){
347  int HitPointIncrement = RangerIterator->second->HitPoints() - Asset->MaxHitPoints();
348 
349  Asset->ChangeType(RangerIterator->second);
350  Asset->IncrementHitPoints(HitPointIncrement);
351  }
352  }
353  }
354  }
355  else if(atBarracks == DActor->Type()){
356  TempEvent.DType = etReady;
357  TempEvent.DAsset = DTarget;
358 
359  DTarget->PopCommand();
360  DTarget->TilePosition(DPlayerData->PlayerMap()->FindAssetPlacement(DTarget, DActor, CPosition(DPlayerData->PlayerMap()->Width()-1, DPlayerData->PlayerMap()->Height()-1)));
361  }
362  DPlayerData->AddGameEvent(TempEvent);
363  DActor->PopCommand();
364  return true;
365  }
366  return false;
367 }
