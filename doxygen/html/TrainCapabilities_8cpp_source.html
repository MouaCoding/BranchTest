135 bool CPlayerCapabilityTrainNormal::CActivatedCapability::IncrementStep(){
136  int AddHitPoints = (DTarget->MaxHitPoints() * (DCurrentStep + 1) / DTotalSteps) - (DTarget->MaxHitPoints() * DCurrentStep / DTotalSteps);
137 
138  DTarget->IncrementHitPoints(AddHitPoints);
139  if(DTarget->HitPoints() > DTarget->MaxHitPoints()){
140  DTarget->HitPoints(DTarget->MaxHitPoints());
141  }
142  DCurrentStep++;
143  DActor->IncrementStep();
144  DTarget->IncrementStep();
145  if(DCurrentStep >= DTotalSteps){
146  SGameEvent TempEvent;
147 
148  TempEvent.DType = etReady;
149  TempEvent.DAsset = DTarget;
150  DPlayerData->AddGameEvent(TempEvent);
151 
152  DTarget->PopCommand();
153  DActor->PopCommand();
154  DTarget->TilePosition(DPlayerData->PlayerMap()->FindAssetPlacement(DTarget, DActor, CPosition(DPlayerData->PlayerMap()->Width()-1, DPlayerData->PlayerMap()->Height()-1)));
155  return true;
156  }
157  return false;
158 }
