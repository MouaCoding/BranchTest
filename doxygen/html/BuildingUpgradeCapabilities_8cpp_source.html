131 bool CPlayerCapabilityBuildingUpgrade::CActivatedCapability::IncrementStep(){
132  int AddHitPoints = ((DUpgradeType->HitPoints() - DOriginalType->HitPoints()) * (DCurrentStep + 1) / DTotalSteps) - ((DUpgradeType->HitPoints() - DOriginalType->HitPoints()) * DCurrentStep / DTotalSteps);
133 
134  if(0 == DCurrentStep){
135  SAssetCommand AssetCommand = DActor->CurrentCommand();
136  AssetCommand.DAction = aaConstruct;
137  DActor->PopCommand();
138  DActor->PushCommand(AssetCommand);
139  DActor->ChangeType(DUpgradeType);
140  DActor->ResetStep();
141  }
142 
143  DActor->IncrementHitPoints(AddHitPoints);
144  if(DActor->HitPoints() > DActor->MaxHitPoints()){
145  DActor->HitPoints(DActor->MaxHitPoints());
146  }
147  DCurrentStep++;
148  DActor->IncrementStep();
149  if(DCurrentStep >= DTotalSteps){
150  SGameEvent TempEvent;
151 
152  TempEvent.DType = etWorkComplete;
153  TempEvent.DAsset = DActor;
154  DPlayerData->AddGameEvent(TempEvent);
155 
156  DActor->PopCommand();
157  if(DActor->Range()){
158  SAssetCommand Command;
159  Command.DAction = aaStandGround;
160  DActor->PushCommand(Command);
161  }
162  return true;
163  }
164  return false;
165 }
