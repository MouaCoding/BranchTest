88  else{
89  DVisibleWidth = MiniMapWidth;

91  }

93  DMapRenderer->DrawMiniMap(DWorkingPixmap);
94  DAssetRenderer->DrawMiniAssets(DWorkingPixmap);
95  if(nullptr != DFogRenderer){
96  DFogRenderer->DrawMiniMap(DWorkingPixmap);
97  }
98 
99  DWorkingPixbuf = gdk_pixbuf_get_from_drawable(DWorkingPixbuf, DWorkingPixmap, nullptr, 0, 0, 0, 0, -1, -1);
100 
101  ScaledPixbuf = gdk_pixbuf_scale_simple(DWorkingPixbuf, DVisibleWidth, DVisibleHeight, GDK_INTERP_BILINEAR);
102  gdk_draw_pixbuf(drawable, TempGC, ScaledPixbuf, 0, 0, 0, 0, -1, -1, GDK_RGB_DITHER_NONE, 0, 0);
103  g_object_unref(ScaledPixbuf);
104 
105  if(nullptr != DViewportRenderer){
106  gdk_gc_set_rgb_fg_color(TempGC, &DViewportColor);
107  MiniMapViewportX = (DViewportRenderer->ViewportX() * DVisibleWidth) / DMapRenderer->DetailedMapWidth();
108  MiniMapViewportY = (DViewportRenderer->ViewportY() * DVisibleHeight) / DMapRenderer->DetailedMapHeight();
109  MiniMapViewportWidth = (DViewportRenderer->LastViewportWidth() * DVisibleWidth) / DMapRenderer->DetailedMapWidth();
110  MiniMapViewportHeight = (DViewportRenderer->LastViewportHeight() * DVisibleHeight) / DMapRenderer->DetailedMapHeight();
111  gdk_draw_rectangle(drawable, TempGC, FALSE, MiniMapViewportX, MiniMapViewportY, MiniMapViewportWidth, MiniMapViewportHeight);
112  }
113 
114  g_object_unref(TempGC);
115 }
