31  int Width, Height, NumberChannels, RowStride;
32 
33  PrintDebug(DEBUG_LOW,"CGraphicMulticolorTileset::LoadTileset\n");
34  if(!CGraphicTileset::LoadTileset(source)){
35  return false;
36  }
37  Height = gdk_pixbuf_get_height(DPixbufTileset);
38  Width = gdk_pixbuf_get_width(DPixbufTileset);
39  NumberChannels = gdk_pixbuf_get_n_channels(DPixbufTileset);
40  RowStride = gdk_pixbuf_get_rowstride(DPixbufTileset);
41 
42  PrintDebug(DEBUG_LOW,"Translating Colors\n");
43  DPixbufTilesets.clear();
44  DPixbufTilesets.push_back(DPixbufTileset);
45  for(int Index = 1; Index < colormap->GroupCount(); Index++){
46  DPixbufTilesets.push_back(gdk_pixbuf_copy(DPixbufTileset));
47 
48  for(int YPos = 0; YPos < Height; YPos++){
49  for(int XPos = 0; XPos < Width; XPos++){
50  guchar *Pixel = gdk_pixbuf_get_pixels(DPixbufTilesets.back()) + YPos * RowStride + XPos * NumberChannels;
51  for(int ColIndex = 0; ColIndex < colormap->ColorCount(); ColIndex++){
52  if(colormap->Recolor(Index, Pixel[0], Pixel[1], Pixel[2])){
53  break;
54  }
55  }
56  }
57  }
58  }
59
