243 bool CPlayerUpgrade::Load(std::shared_ptr< CDataSource > source){
244  CLineDataSource LineSource(source);
245  std::string Name, TempString;
246  std::shared_ptr< CPlayerUpgrade > PlayerUpgrade;
247  EAssetCapabilityType UpgradeType;
248  int AffectedAssetCount;
249  bool ReturnStatus = false;
250 
251  if(nullptr == source){
252  return false;
253  }
254  if(!LineSource.Read(Name)){
255  PrintError("Failed to get player upgrade name.\n");
256  return false;
257  }
258  UpgradeType = CPlayerCapability::NameToType(Name);
259  if((actNone == UpgradeType) && (Name != CPlayerCapability::TypeToName(actNone))){
260  PrintError("Unknown upgrade type %s.\n", Name.c_str());
261  return false;
262  }
263  auto Iterator = DRegistryByName.find(Name);
264  if(DRegistryByName.end() != Iterator){
265  PlayerUpgrade = Iterator->second;
266  }
267  else{
268  PlayerUpgrade = std::make_shared< CPlayerUpgrade >();
269  PlayerUpgrade->DName = Name;
270  DRegistryByName[Name] = PlayerUpgrade;
271  DRegistryByType[UpgradeType] = PlayerUpgrade;
272  }
273  try{
274  if(!LineSource.Read(TempString)){
275  PrintError("Failed to get upgrade armor.\n");
276  goto LoadExit;
277  }
278  PlayerUpgrade->DArmor = std::stoi(TempString);
279  if(!LineSource.Read(TempString)){
280  PrintError("Failed to get upgrade sight.\n");
281  goto LoadExit;
282  }
283  PlayerUpgrade->DSight = std::stoi(TempString);
284  if(!LineSource.Read(TempString)){
285  PrintError("Failed to get upgrade speed.\n");
286  goto LoadExit;
287  }
288  PlayerUpgrade->DSpeed = std::stoi(TempString);
289  if(!LineSource.Read(TempString)){
290  PrintError("Failed to get upgrade basic damage.\n");
291  goto LoadExit;
292  }
293  PlayerUpgrade->DBasicDamage = std::stoi(TempString);
294  if(!LineSource.Read(TempString)){
295  PrintError("Failed to get upgrade piercing damage.\n");
296  goto LoadExit;
297  }
298  PlayerUpgrade->DPiercingDamage = std::stoi(TempString);
299  if(!LineSource.Read(TempString)){
300  PrintError("Failed to get upgrade range.\n");
301  goto LoadExit;
302  }
303  PlayerUpgrade->DRange = std::stoi(TempString);
304  if(!LineSource.Read(TempString)){
305  PrintError("Failed to get upgrade gold cost.\n");
306  goto LoadExit;
307  }
308  PlayerUpgrade->DGoldCost = std::stoi(TempString);
309  if(!LineSource.Read(TempString)){
310  PrintError("Failed to get upgrade lumber cost.\n");
311  goto LoadExit;
312  }
313  PlayerUpgrade->DLumberCost = std::stoi(TempString);
314  if(!LineSource.Read(TempString)){
315  PrintError("Failed to get upgrade research time.\n");
316  goto LoadExit;
317  }
318  PlayerUpgrade->DResearchTime = std::stoi(TempString);
319  if(!LineSource.Read(TempString)){
320  PrintError("Failed to get upgrade affected asset count.\n");
321  goto LoadExit;
322  }
323  AffectedAssetCount = std::stoi(TempString);
324  for(int Index = 0; Index < AffectedAssetCount; Index++){
325  if(!LineSource.Read(TempString)){
326  PrintError("Failed to read upgrade affected asset %d.\n", Index);
327  goto LoadExit;
328  }
329  PlayerUpgrade->DAffectedAssets.push_back(CPlayerAssetType::NameToType(TempString));
330  }
331  ReturnStatus = true;
332  }
333  catch(std::exception &E){
334  PrintError("%s\n",E.what());
335  }
336 LoadExit:
337  return ReturnStatus;
338 }

784 bool CPlayerAsset::MoveStep(std::vector< std::vector< std::shared_ptr< CPlayerAsset > > > &occupancymap, std::vector< std::vector< bool > > &diagonals){
785  EDirection CurrentOctant = DPosition.TileOctant();
786  const int DeltaX[] = {0, 5, 7, 5, 0, -5, -7, -5};
787  const int DeltaY[] = {-7, -5, 0, 5, 7, 5, 0, -5};
788  CPosition CurrentTile(DTilePosition), CurrentPosition(DPosition);
789 
790  if((dMax == CurrentOctant)||(CurrentOctant == DDirection)){// Aligned just move
791  int NewX = Speed() * DeltaX[DDirection] * CPosition::TileWidth() + DMoveRemainderX;
792  int NewY = Speed() * DeltaY[DDirection] * CPosition::TileHeight() + DMoveRemainderY;
793  DMoveRemainderX = NewX % DUpdateDivisor;
794  DMoveRemainderY = NewY % DUpdateDivisor;
795  DPosition.IncrementX(NewX / DUpdateDivisor);
796  DPosition.IncrementY(NewY / DUpdateDivisor);
797  }
798  else{ // Entering
799  int NewX = Speed() * DeltaX[DDirection] * CPosition::TileWidth() + DMoveRemainderX;
800  int NewY = Speed() * DeltaY[DDirection] * CPosition::TileHeight() + DMoveRemainderY;
801  int TempMoveRemainderX = NewX % DUpdateDivisor;
802  int TempMoveRemainderY = NewY % DUpdateDivisor;
803  CPosition NewPosition(DPosition.X() + NewX / DUpdateDivisor, DPosition.Y() + NewY / DUpdateDivisor);
804 
805  if(NewPosition.TileOctant() == DDirection){
806  NewPosition.SetToTile(NewPosition);
807  NewPosition.SetFromTile(NewPosition);
808  TempMoveRemainderX = TempMoveRemainderY = 0;
809  }
810  DPosition = NewPosition;
811  DMoveRemainderX = TempMoveRemainderX;
812  DMoveRemainderY = TempMoveRemainderY;
813  }
814  DTilePosition.SetToTile(DPosition);
815  if(CurrentTile != DTilePosition){
816  bool Diagonal = (CurrentTile.X() != DTilePosition.X()) && (CurrentTile.Y() != DTilePosition.Y());
817  int DiagonalX = MIN(CurrentTile.X(), DTilePosition.X());
818  int DiagonalY = MIN(CurrentTile.Y(), DTilePosition.Y());
819 
820  if(occupancymap[DTilePosition.Y()][DTilePosition.X()] || (Diagonal && diagonals[DiagonalY][DiagonalX])){
821  bool ReturnValue = false;
822  if(aaWalk == occupancymap[DTilePosition.Y()][DTilePosition.X()]->Action()){
823  ReturnValue = occupancymap[DTilePosition.Y()][DTilePosition.X()]->Direction() == CurrentPosition.TileOctant();
824  }
825  DTilePosition = CurrentTile;
826  DPosition = CurrentPosition;
827  return ReturnValue;
828  }
829  if(Diagonal){
830  diagonals[DiagonalY][DiagonalX] = true;
831  }
832  occupancymap[DTilePosition.Y()][DTilePosition.X()] = occupancymap[CurrentTile.Y()][CurrentTile.X()];
833  occupancymap[CurrentTile.Y()][CurrentTile.X()] = nullptr;
834  }
835 
836  IncrementStep();
837  return true;
838 }
