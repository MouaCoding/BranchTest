68 void CVisibilityMap::Update(const std::list< std::weak_ptr< CPlayerAsset > > &assets){
69  for(auto &Row : DMap){
70  for(auto &Cell : Row){
71  if((tvVisible == Cell)||(tvPartial == Cell)){
72  Cell = tvSeen;
73  }
74  else if(tvPartialPartial == Cell){
75  Cell = tvSeenPartial;
76  }
77  }
78  }
79  for(auto WeakAsset : assets){
80  if(auto CurAsset = WeakAsset.lock()){
81  CPosition Anchor = CurAsset->TilePosition();
82  int Sight = CurAsset->EffectiveSight() + CurAsset->Size()/2;
83  int SightSquared = Sight * Sight;
84  Anchor.X(Anchor.X() + CurAsset->Size()/2);
85  Anchor.Y(Anchor.Y() + CurAsset->Size()/2);
86  for(int X = 0; X <= Sight; X++){
87  int XSquared = X * X;
88  int XSquared1 = X ? (X - 1) * (X - 1) : 0;
89 
90  for(int Y = 0; Y <= Sight; Y++){
91  int YSquared = Y * Y;
92  int YSquared1 = Y ? (Y - 1) * (Y - 1) : 0;
93 
94  if((XSquared + YSquared) < SightSquared){
95  // Visible
96  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvVisible;
97  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvVisible;
98  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvVisible;
99  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvVisible;
100  }
101  else if((XSquared1 + YSquared1) < SightSquared){
102  // Partial
103  ETileVisibility CurVis = DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility];
104  if(tvSeen == CurVis){
105  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvPartial;
106  }
107  else if((tvNone == CurVis)||(tvSeenPartial == CurVis)){
108  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvPartialPartial;
109  }
110  CurVis = DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility];
111  if(tvSeen == CurVis){
112  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvPartial;
113  }
114  else if((tvNone == CurVis)||(tvSeenPartial == CurVis)){
115  DMap[Anchor.Y() - Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvPartialPartial;
116  }
117  CurVis = DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility];
118  if(tvSeen == CurVis){
119  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvPartial;
120  }
121  else if((tvNone == CurVis)||(tvSeenPartial == CurVis)){
122  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() - X + DMaxVisibility] = tvPartialPartial;
123  }
124  CurVis = DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility];
125  if(tvSeen == CurVis){
126  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvPartial;
127  }
128  else if((tvNone == CurVis)||(tvSeenPartial == CurVis)){
129  DMap[Anchor.Y() + Y + DMaxVisibility][Anchor.X() + X + DMaxVisibility] = tvPartialPartial;
130  }
131  }
132  }
133  }
134  }
135  }
136  int MinX, MinY, MaxX, MaxY;
137  MinY = DMaxVisibility;
138  MaxY = DMap.size() - DMaxVisibility;
139  MinX = DMaxVisibility;
140  MaxX = DMap[0].size() - DMaxVisibility;
141  DUnseenTiles = 0;
142  for(int Y = MinY; Y < MaxY; Y++){
143  for(int X = MinX; X < MaxX; X++){
144  if(tvNone == DMap[Y][X]){
145  DUnseenTiles++;
146  }
147  }
148  }
