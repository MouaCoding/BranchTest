42 void CPosition::SetTileDimensions(int width, int height){

44  DTileWidth = width;
45  DTileHeight = height;
46  DHalfTileWidth = width / 2;
47  DHalfTileHeight = height / 2;
48 
49  DOctant.resize(DTileHeight);
50  for(auto &Row : DOctant){
51  Row.resize(DTileWidth);
52  }
53  for(int Y = 0; Y < DTileHeight; Y++){
54  for(int X = 0; X < DTileWidth; X++){
55  int XDistance = X - DHalfTileWidth;
56  int YDistance = Y - DHalfTileHeight;
57  bool NegativeX = XDistance < 0;
58  bool NegativeY = YDistance > 0; // Top of screen is 0
59  double SinSquared;
60 
61  XDistance *= XDistance;
62  YDistance *= YDistance;
63 
64  if(0 == (XDistance + YDistance)){
65  DOctant[Y][X] = dMax;
66  continue;
67  }
68  SinSquared = (double)YDistance / (XDistance + YDistance);
69 
70  if(0.1464466094 > SinSquared){
71  // East or West
72  if(NegativeX){
73  DOctant[Y][X] = dWest; // West
74  }
75  else{
76  DOctant[Y][X] = dEast; // East
77  }
78  }
79  else if(0.85355339059 > SinSquared){
80  // NE, SE, SW, NW
81  if(NegativeY){
82  if(NegativeX){
83  DOctant[Y][X] = dSouthWest; // SW
84  }

174 CPosition CPosition::ClosestPosition(const CPosition &objpos, int objsize) const{
175  CPosition CurPosition(objpos);

194  CPosition DeltaPosition(pos.DX - DX, pos.DY - DY);
195  int DivX = DeltaPosition.DX / HalfTileWidth();
196  int DivY = DeltaPosition.DY / HalfTileHeight();

201

204  DeltaPosition.DY /= Div;
205  }
206  DeltaPosition.DX += HalfTileWidth();
207  DeltaPosition.DY += HalfTileHeight();
208  if(0 > DeltaPosition.DX){

215  DeltaPosition.DX = TileWidth() - 1;

230 int CPosition::Distance(const CPosition &pos){
231  // Code from http://www.codecodex.com/wiki/Calculate_an_integer_square_root
232  unsigned int Op, Result, One;
233 
234  Op = DistanceSquared(pos);
235  Result = 0;
236 
237  One = 1 << (sizeof(unsigned int) * 8 - 2);
238  while(One > Op){
239  One >>= 2;
240  }
241  while(0 != One){
242  if(Op >= Result + One){
243  Op -= Result + One;
244  Result += One << 1; // <-- faster than 2 * one
245  }
246  Result >>= 1;
247  One >>= 2;
248  }
249  return Result;
