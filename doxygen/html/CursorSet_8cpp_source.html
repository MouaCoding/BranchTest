35 bool CCursorSet::LoadCursors(std::shared_ptr< CDataSource > source){
36  CLineDataSource LineSource(source);
37  std::string TempString;
38  std::vector< std::string > Tokens;
39  bool ReturnStatus = false;
40 
41  if(nullptr == source){
42  return false;
43  }
44  if(!CGraphicTileset::LoadTileset(source)){
45  return false;
46  }
47 
48  try{
49  for(int Index = 0; Index < DTileCount; Index++){
50  if(!LineSource.Read(TempString)){
51  PrintError("Failed to get %d cursor point.\n", Index);
52  goto LoadCursorsExit;
53  }
54  CTokenizer::Tokenize(Tokens, TempString);
55  if(2 != Tokens.size()){
56  PrintError("Invalid cursor points %s (%zd).\n",TempString.c_str(), Tokens.size());
57  for(int I = 0; I < Tokens.size(); I++){
58  PrintError("Tokens[%d] = %s\n",I,Tokens[I].c_str());
59  }
60  goto LoadCursorsExit;
61  }
62  DCursorXPoint.push_back(std::stoi(Tokens[0]));
63  DCursorYPoint.push_back(std::stoi(Tokens[1]));
64  }
65  }
66  catch(std::exception &E){
67  PrintError("%s\n",E.what());
68  }
69  ReturnStatus = true;
70 LoadCursorsExit:
71  return ReturnStatus;
72 }
