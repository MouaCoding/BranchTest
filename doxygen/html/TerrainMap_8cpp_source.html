64 CPosition CTerrainMap::FindNearestTileType(const CPosition &pos, ETileType type){
65  int MaxDistance = Width() > Height() ? Width() : Height();
66  int XOffset = pos.X() + 1;
67  int YOffset = pos.Y() + 1;
68 
69  for(int SearchDistance = 0; SearchDistance < MaxDistance; SearchDistance++){
70  int PositiveX = XOffset + SearchDistance;
71  int NegativeX = XOffset - SearchDistance;
72  int PositiveY = YOffset + SearchDistance;
73  int NegativeY = YOffset - SearchDistance;
74  bool SearchPX = true;
75  bool SearchNX = true;
76  bool SearchPY = true;
77  bool SearchNY = true;
78 
79  if(0 >= NegativeX){
80  NegativeX = 1;
81  SearchNX = false;
82  }
83  if(PositiveX + 1 >= DMap[0].size()){
84  PositiveX = DMap[0].size() - 2;
85  SearchPX = false;
86  }
87  if(0 >= NegativeY){
88  NegativeY = 1;
89  SearchNY = false;
90  }
91  if(PositiveY + 1 >= DMap.size()){
92  PositiveY = DMap.size() - 2;
93  SearchPY = false;
94  }
95  if(!SearchNX && !SearchNY && !SearchPX && !SearchPY){
96  break;
97  }
98  if(SearchNY){
99  for(int XOff = NegativeX; XOff <= PositiveX; XOff++){
100  if(type == DMap[NegativeY][XOff]){
101  return CPosition(XOff-1, NegativeY-1);
102  }
103  }
104  }
105  if(SearchPX){
106  for(int YOff = NegativeY; YOff <= PositiveY; YOff++){
107  if(type == DMap[YOff][PositiveX]){
108  return CPosition(PositiveX-1, YOff-1);
109  }
110  }
111  }
112  if(SearchPY){
113  for(int XOff = PositiveX; XOff >= NegativeX; XOff--){
114  if(type == DMap[PositiveY][XOff]){
115  return CPosition(XOff-1, PositiveY-1);
116  }
117  }
118  }
119  if(SearchNX){
120  for(int YOff = PositiveY; YOff >= NegativeY; YOff--){
121  if(type == DMap[YOff][NegativeX]){
122  return CPosition(NegativeX-1, YOff-1);
123  }
124  }
125  }
126  }
127  return CPosition(-1, -1);
128 }

143 bool CTerrainMap::LoadMap(std::shared_ptr< CDataSource > source){
144  CLineDataSource LineSource(source);
145  std::string TempString;
146  std::vector< std::string > Tokens;
147  int MapWidth, MapHeight;
148  bool ReturnStatus = false;
149  //int PlayerColorsFound = 0;
150 
151  DMap.clear();
152 
153 
154  if(!LineSource.Read(DMapName)){
155  PrintError("Failed to read map line.\n");
156  goto LoadMapExit;
157  }
158 
159  if(!LineSource.Read(TempString)){
160  PrintError("Failed to read map line.\n");
161  goto LoadMapExit;
162  }
163  CTokenizer::Tokenize(Tokens, TempString);
164  if(2 != Tokens.size()){
165  PrintError("Invalid map dimensions %s (%zd).\n",TempString.c_str(), Tokens.size());
166  for(int I = 0; I < Tokens.size(); I++){
167  PrintError("Tokens[%d] = %s\n",I,Tokens[I].c_str());
168  }
169  goto LoadMapExit;
170  }
171  try{
172  MapWidth = std::stoi(Tokens[0]);
173  MapHeight = std::stoi(Tokens[1]);
174 
175  PrintDebug(DEBUG_LOW, "Map %s is %d x %d\n",DMapName.c_str(), MapWidth, MapHeight);
176  if((8 > MapWidth)||(8 > MapHeight)){
177  PrintError("Invalid map dimensions.\n");
178  goto LoadMapExit;
179  }
180  while(DStringMap.size() < MapHeight + 2){
181  if(!LineSource.Read(TempString)){
182  PrintError("Failed to read map line.\n");
183  goto LoadMapExit;
184  }
185  DStringMap.push_back(TempString);
186  if(MapWidth + 2 > DStringMap.back().length()){
187  PrintError("Map line %zd too short!\n", DStringMap.size());
188  goto LoadMapExit;
189  }
190  }
191  if(MapHeight + 2 > DStringMap.size()){
192  PrintError("Map has too few lines!\n");
193  goto LoadMapExit;
194  }
195  DMap.resize(MapHeight+2);
196  for(int Index = 0; Index < DMap.size(); Index++){
197  DMap[Index].resize(MapWidth+2);
198  for(int Inner = 0; Inner < MapWidth+2; Inner++){
199  switch(DStringMap[Index][Inner]){
200  case 'G': DMap[Index][Inner] = ttGrass;
201  break;
202  case 'F': DMap[Index][Inner] = ttTree;
203  break;
204  case 'D': DMap[Index][Inner] = ttDirt;
205  break;
206  case 'W': DMap[Index][Inner] = ttWall;
207  break;
208  case 'w': DMap[Index][Inner] = ttWallDamaged;
209  break;
210  case 'R': DMap[Index][Inner] = ttRock;
211  break;
212  case ' ': DMap[Index][Inner] = ttWater;
213  break;
214  default: PrintError("Unknwon tile type '%c'on line %d\n", DStringMap[Index][Inner],Index+2);
215  goto LoadMapExit;
216  break;
217  }
218  }
219  }
220  ReturnStatus = true;
221  }
222  catch(std::exception &E){
223  PrintError("%s\n",E.what());
224  }
225 
226 LoadMapExit:
227  return ReturnStatus;
228 }
