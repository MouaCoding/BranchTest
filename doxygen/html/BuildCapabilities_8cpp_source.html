109 bool CPlayerCapabilityBuildNormal::ApplyCapability(std::shared_ptr< CPlayerAsset > actor, std::shared_ptr< CPlayerData > playerdata, std::shared_ptr< CPlayerAsset > target){
110  auto Iterator = playerdata->AssetTypes()->find(DBuildingName);
111 
112  if(Iterator != playerdata->AssetTypes()->end()){
113  SAssetCommand NewCommand;
114 
115  actor->ClearCommand();
116  if(actor->TilePosition() == target->TilePosition()){
117  auto AssetType = Iterator->second;
118  auto NewAsset = playerdata->CreateAsset(DBuildingName);
119  CPosition TilePosition;
120  TilePosition.SetToTile(target->Position());
121  NewAsset->TilePosition(TilePosition);
122  NewAsset->HitPoints(1);
123 
124  NewCommand.DAction = aaCapability;
125  NewCommand.DCapability = AssetCapabilityType();
126  NewCommand.DAssetTarget = NewAsset;
127  NewCommand.DActivatedCapability = std::make_shared< CActivatedCapability >(actor, playerdata, NewAsset, AssetType->LumberCost(), AssetType->GoldCost(), CPlayerAsset::UpdateFrequency() * AssetType->BuildTime());
128  actor->PushCommand(NewCommand);
129  }
130  else{
131  NewCommand.DAction = aaCapability;
132  NewCommand.DCapability = AssetCapabilityType();
133  NewCommand.DAssetTarget = target;
134  actor->PushCommand(NewCommand);
135 
136  NewCommand.DAction = aaWalk;
137  actor->PushCommand(NewCommand);
138  }
139  return true;

164 bool CPlayerCapabilityBuildNormal::CActivatedCapability::IncrementStep(){
165  int AddHitPoints = (DTarget->MaxHitPoints() * (DCurrentStep + 1) / DTotalSteps) - (DTarget->MaxHitPoints() * DCurrentStep / DTotalSteps);
166 
167  DTarget->IncrementHitPoints(AddHitPoints);
168  if(DTarget->HitPoints() > DTarget->MaxHitPoints()){
169  DTarget->HitPoints(DTarget->MaxHitPoints());
170  }
171  DCurrentStep++;
172  DActor->IncrementStep();
173  DTarget->IncrementStep();
174  if(DCurrentStep >= DTotalSteps){
175  SGameEvent TempEvent;
176 
177  TempEvent.DType = etWorkComplete;
178  TempEvent.DAsset = DActor;
179  DPlayerData->AddGameEvent(TempEvent);
180 
181  DTarget->PopCommand();
182  DActor->PopCommand();
183  DActor->TilePosition(DPlayerData->PlayerMap()->FindAssetPlacement(DActor, DTarget, CPosition(DPlayerData->PlayerMap()->Width()-1, DPlayerData->PlayerMap()->Height()-1)));
184  DActor->ResetStep();
185  DTarget->ResetStep();
186 
187  return true;
188  }
189  return false;
190 }
